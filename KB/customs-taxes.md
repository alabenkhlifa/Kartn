### Price Estimation Algorithm

```python
def estimate_final_price(cif, engine_cc, fuel_type, regime, is_electric, is_phev, is_hev):
    # Droits de Douane (assuming EU origin = 0%)
    dd_rate = 0 if eu_origin else 0.20
    if is_electric or is_phev:
        dd_rate = 0

    # Taxe de Consommation
    if is_electric or is_phev:
        dc_rate = 0
    elif regime == "FCR_FAMILLE":
        dc_rate = 0.10  # Article 55
    elif is_hev and engine_cc <= 1700:
        dc_rate = get_dc_rate(engine_cc, fuel_type) * 0.5
    else:
        dc_rate = get_dc_rate(engine_cc, fuel_type)

    # TVA
    if is_electric or is_phev or regime == "FCR_FAMILLE":
        tva_rate = 0.07
    else:
        tva_rate = 0.19

    # Calculate
    dd = cif * dd_rate
    dc = (cif + dd) * dc_rate
    tva = (cif + dd + dc) * tva_rate
    tfd = (dd + dc) * 0.03

    total_taxes = dd + dc + tva + tfd
    final_price = cif + total_taxes

    # FCR TRE = 25% of total taxes
    if regime == "FCR_TRE":
        final_price = cif + (total_taxes * 0.25)

    return final_price
```
